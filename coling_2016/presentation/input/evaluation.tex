\begin{frame}{Evaluation}\framesubtitle{Measures}
  Stem-based comparisons at the top 10 outputed keyphrases
  \begin{itemize}
    \item{Recall}%: true positive maximization}
    \item{Precision}%: false positive minimization}
    \item{F1-score}%: recall and precision trade-off}
  \end{itemize}
\end{frame}
